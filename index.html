const amountIn = document.getElementById('amount'); const fromCr = document.getElementById('fromCurrency'); const toCr = document.getElementById('toCurrency'); const resDisplay = document.getElementById('result'); const unitRateEl = document.getElementById('unitRate'); const swapBtn = document.getElementById('swapBtn'); async function calculate() { const from = fromCr.value; const to = toCr.value; const amount = parseFloat(amountIn.value); if (!amount || amount <= 0) { resDisplay.innerText = "0.00"; unitRateEl.innerText = "Enter amount"; return; } // Показываем пользователю, что запрос идет (UX) resDisplay.style.opacity = "0.5"; unitRateEl.innerText = "Fetching data..."; try { // Используем HTTPS и добавляем таймштамп для обхода кэша const url = `https://min-api.cryptocompare.com{from.toUpperCase()}&tsyms=${to.toUpperCase()}&extraParams=smartcalc_app_${Date.now()}`; const response = await fetch(url); if (!response.ok) throw new Error('Network response was not ok'); const data = await response.json(); if (data && data[to.toUpperCase()]) { const rate = data[to.toUpperCase()]; const total = amount * rate; // Форматирование для профи: если число маленькое - 6 знаков, если большое - 2 resDisplay.innerText = total > 1 ? total.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 }) : total.toFixed(6); unitRateEl.innerText = `1 ${from} = ${rate.toLocaleString('en-US')} ${to}`; } else { throw new Error('Invalid Data'); } } catch (err) { console.error("API Error:", err); resDisplay.innerText = "Offline"; unitRateEl.innerText = "Service temporarily unavailable"; } finally { resDisplay.style.opacity = "1"; } } // Слушатели с защитой от "дребезга" (debounce) let timeout; [amountIn, fromCr, toCr].forEach(el => { el.addEventListener('input', () => { clearTimeout(timeout); timeout = setTimeout(calculate, 300); // Задержка 300мс, чтобы не спамить API }); }); swapBtn.addEventListener('click', () => { const temp = fromCr.value; fromCr.value = toCr.value; toCr.value = temp; calculate(); }); // Стартовый запуск calculate();
